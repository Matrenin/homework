let ulEl = document.querySelector("ul");
let navLinks = document.querySelectorAll("a");
ulEl.addEventListener("click", event => {
  if (event.target.tagName == "A") {
    for (let link of navLinks) {
      if (link == event.target) {
        link.classList.add("active");
      } else {
        link.classList.remove("active");
      }
    }
  }
});

const panelText = {
    T1: "Уверен, ты справишься с этим заданием!",
    X6: "Знаю, программирование это нелегко.",
    H99: "Однако, программирование, в итоге, принесет свои плоды.",
 };

let liEls = document.querySelectorAll("li");
let divText = document.querySelector(".text");
liEls.forEach(li => {
  li.addEventListener("click", event => {
    let text = event.currentTarget.dataset.text;
    divText.innerText = panelText[text];
  });
});
