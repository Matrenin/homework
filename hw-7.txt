let express = require("express");   // подключаем экспресс
let fs = require("fs");   // подключаем фс, чтобы считывать данные

let app = express();

// use - настройка middleware
app.use(express.json());    // говорим, что наш express будет работать с форматом json
app.use("/", express.static, "/public");    // запросы к нашему корневому url("/"), мы будет одавать статичные данные из папки "/public"

// описываем процесс, который будет происходить с нашей api
app.get("/api/products", (req, res) => {    // запрос на url "/api/products" 
  fs.readFile("./server/db/products.json", "utf-8", (err, data) => {   // считываем файл из папки "./server/db/products.json"
     if (err) res.send(JSON.strigify({result: 0, err}));
     else res.send(data);
  });
});

app.post("/api/cart", (req, res) => {
  fs.readFile("./server/db/userCart.json", "utf-8", (err, data) => {
     if (err) res.send(JSON.strigify({result: 0, err}));
     else {
       let cart = JSON.parse(data);
       cart.contents.push(req.body);
      
       fs.writeFile("./server/db/userCart.json", JSON.stringify(cart), err => {
         if (err) res.send(JSON.strigify({result: 0, err}));
         else res.send(JSON.strigify({result: 1, err}));
      });
    }
  });
});

app.put("/api/cart/:id", (req, res) => {    // добавить данные, ":id" - когда мы ставим : это становится параметром фрейворка
  fs.readFile("./server/db/userCart.json", "utf-8", (err, data) => {
    if (err) res.send(JSON.strigify({result: 0, err}));
    else {
      let cart = JSON.parse(data);
      let find = cart.contents.find(el => el.id_product === Number(req.params.id));     // через свойство params могу получить данные 
      find.quantity += req.body.quantity;   // меняем значения
      
      fs.writeFile("./server/db/userCart.json", JSON.strigify(cart), (err) => {   // записываем новые значения
         if (err) res.send(JSON.strigify({result: 0, err}));
         else res.send(JSON.strigify({result: 1}));
      });
    }
  });
});

app.Listen(5555, () => {
  console.log("Server started");
});

Данные которые мы хотим передать на сервер всегда запаковываются в body.
