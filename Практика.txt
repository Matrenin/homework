let getRequest = (url, cb) => {
  let xhr = new XMLHttpRequest();	// создаем экземпляр.
  xhr.open("GET", url, true);		// конфигурируем запрос.
  xhr.onreadystatechange = () => {
    if (xhr.readyState === 4) {
      if (xhr.status !== 200) {
        console.log(err);
      } else {
        cb(xhr.responseText);
      }
    }
  };
  xhr.send();
};

> getRequest - функция которая делает запрос.
url - куда мы передаем запрос. 
cd - функция которая будет срабатывать по завершению нашего запроса. По истечению времени она будет вызвана со значением xhr.responseText.

onreadystatechange - свойство ловит момент, когда ответ от сервиса получен.
readyState - описывает состояние загрузки.
status - по коду можно понять, что ответил сервер.
